This dataset is from New Yorks City Council. It displays the statistics for the recommended special education services and other related services by students who are fully receiving, partially receiving, or not receiving services at all. Moreover, the dataset is composed of program type, superintendent, district and school.

library(tidyverse)
library(dplyr)

ccreport2021 <- ccreport2021 |>
  rename(primpprogramtype = `Primary Program Type`,
         fully_receiving = `Fully Receiving`,
         fully_receivingperc = `Percent Fully Receiving`,
         partially_receiving = `Partially Receiving`,
         partially_receivingperc = `Percent Partially Receiving`,
         not_receiving = `Not Receiving`,
         not_receivingperc = `Percent Not Receiving`) |>
  filter(primpprogramtype == c("Integrated Co-Teaching Services", "SETSS", "Special Class"))
